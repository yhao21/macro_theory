\section{Taylor Exansion}
\begin{align*}
f(x)&= \sum\limits_{n = 0} ^\infty \frac{f^{(n)}(a)(x - a)^{n}}{n!}\\
&= \frac{f(a)\cdot 1}{1} + f'(a)(x - a) + \frac{1}{2}f''(a)(x - a)^{2} + ...
\end{align*}
where
\begin{equation*}
f^{(0)}(a) = f(a), \quad 0! = 1
\end{equation*}
